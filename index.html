Our Flask app


                            app = Flask('__name__')

                            @app.route('/api/time')
                            def time():
                                return jsonify(now=datetime.now().isoformat())


                            @app.route('/api/greet/<name>')
                            @app.route('/api/greet-stranger/', defaults={'name': 'mysterious person'})
                            def greeting(name):
                                msg = 'Welcome, ' + name
                                return jsonify(greeting=msg)
